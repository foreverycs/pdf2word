  <script>
    const modeTabs = document.querySelectorAll('.mode-tab');
    const input = document.getElementById('input');
    const output = document.getElementById('output');
    const indent = document.getElementById('indent');
    const indentField = document.getElementById('indentField');
    const sortKeys = document.getElementById('sortKeys');
    const ensureAscii = document.getElementById('ensureAscii');
    const btnRun = document.getElementById('btnRun');
    const btnValidate = document.getElementById('btnValidate');
    const btnSwap = document.getElementById('btnSwap');
    const btnCopy = document.getElementById('btnCopy');
    const btnSample = document.getElementById('btnSample');
    const btnClearIn = document.getElementById('btnClearIn');
    const btnClearOut = document.getElementById('btnClearOut');
    const meta = document.getElementById('meta');
    const status = document.getElementById('status');

    let mode = 'pretty';

    function setStatus(kind, text) {
      status.className = 'status show ' + kind;
      status.textContent = text;
    }
    function clearStatus() {
      status.className = 'status';
      status.textContent = '';
    }

    function applyMode(next) {
      mode = next;
      modeTabs.forEach((t) => {
        t.classList.toggle('active', t.getAttribute('data-mode') === mode);
      });
      if (mode === 'pretty') {
        btnRun.textContent = '格式化';
        indentField.classList.remove('is-inactive');
        indentField.setAttribute('aria-hidden', 'false');
      } else {
        btnRun.textContent = '压缩';
        indentField.classList.add('is-inactive');
        indentField.setAttribute('aria-hidden', 'true');
      }
      clearStatus();
    }

    modeTabs.forEach((tab) => {
      tab.addEventListener('click', () => applyMode(tab.getAttribute('data-mode')));
    });
    applyMode('pretty');

    btnClearIn.addEventListener('click', () => { input.value = ''; });
    btnClearOut.addEventListener('click', () => { output.value = ''; meta.innerHTML = ''; });

    btnSample.addEventListener('click', () => {
      input.value = JSON.stringify({
        name: '工具集',
        version: 1,
        features: ['json', 'base64'],
        nested: { ok: true, count: 2 },
        empty: null
      });
    });

    btnSwap.addEventListener('click', () => {
      if (!output.value) return;
      input.value = output.value;
    });

    btnCopy.addEventListener('click', async () => {
      const text = output.value;
      if (!text) return;
      try {
        await navigator.clipboard.writeText(text);
        setStatus('ok', '已复制到剪贴板');
      } catch (e) {
        output.select();
        document.execCommand('copy');
        setStatus('ok', '已复制');
      }
    });

    function showMeta(obj) {
      const parts = [];
      if (obj.type) parts.push(`类型 <strong>${obj.type}</strong>`);
      if (obj.mode) parts.push(`模式 <strong>${obj.mode}</strong>`);
      if (obj.indent != null && obj.mode === 'pretty') parts.push(`缩进 <strong>${obj.indent}</strong>`);
      if (obj.input_chars != null) parts.push(`输入 <strong>${obj.input_chars}</strong> 字符`);
      if (obj.output_chars != null) parts.push(`输出 <strong>${obj.output_chars}</strong> 字符`);
      if (obj.sort_keys) parts.push(`键排序 <strong>开</strong>`);
      if (obj.ensure_ascii) parts.push(`ASCII <strong>开</strong>`);
      meta.innerHTML = parts.join(' · ');
    }

    function detailMessage(data) {
      const detail = data.detail;
      if (typeof detail === 'string') return detail;
      if (detail && typeof detail === 'object' && !Array.isArray(detail)) {
        let msg = detail.message || detail.msg || '请求失败';
        if (detail.line != null) {
          msg += `（第 ${detail.line} 行，第 ${detail.column} 列）`;
        }
        return msg;
      }
      if (Array.isArray(detail)) {
        return detail.map((d) => d.msg || d.message || d).join('; ');
      }
      return data.message || `HTTP 错误`;
    }

    async function runFormat() {
      clearStatus();
      btnRun.disabled = true;
      try {
        const fd = new FormData();
        fd.append('text', input.value);
        fd.append('mode', mode);
        fd.append('indent', indent.value);
        fd.append('sort_keys', sortKeys.checked ? 'true' : 'false');
        fd.append('ensure_ascii', ensureAscii.checked ? 'true' : 'false');
        const res = await fetch((window.__ROOT__||'')+'/tools/json/format', { method: 'POST', body: fd });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(detailMessage(data));
        output.value = data.result || '';
        setStatus('ok', mode === 'pretty' ? '格式化完成' : '压缩完成');
        showMeta(data);
      } catch (err) {
        setStatus('err', err.message || String(err));
      } finally {
        btnRun.disabled = false;
      }
    }

    async function runValidate() {
      clearStatus();
      try {
        const fd = new FormData();
        fd.append('text', input.value);
        const res = await fetch((window.__ROOT__||'')+'/tools/json/validate', { method: 'POST', body: fd });
        const data = await res.json().catch(() => ({}));
        if (!res.ok) throw new Error(detailMessage(data));
        if (data.valid) {
          setStatus('ok', `JSON 有效（类型：${data.type || 'value'}）`);
          meta.innerHTML = data.input_chars != null
            ? `输入 <strong>${data.input_chars}</strong> 字符`
            : '';
        } else {
          let msg = data.error || 'JSON 无效';
          if (data.line != null) msg += `（第 ${data.line} 行，第 ${data.column} 列）`;
          setStatus('err', msg);
        }
      } catch (err) {
        setStatus('err', err.message || String(err));
      }
    }

    btnRun.addEventListener('click', runFormat);
    btnValidate.addEventListener('click', runValidate);

    input.addEventListener('keydown', (e) => {
      if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {
        e.preventDefault();
        runFormat();
      }
    });
  </script>
